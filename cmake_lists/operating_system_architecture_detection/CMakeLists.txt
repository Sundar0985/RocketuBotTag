if(NOT DEFINED OPERATING_SYSTEM_ARCHITECTURE)
    if(CMAKE_SYSTEM_PROCESSOR STREQUAL "x86_64" OR
       CMAKE_SYSTEM_PROCESSOR STREQUAL "x64"    OR
       CMAKE_SYSTEM_PROCESSOR STREQUAL "AMD64"  OR
       CMAKE_SYSTEM_PROCESSOR STREQUAL "amd64")
        set(OPERATING_SYSTEM_ARCHITECTURE "x64")
    elseif(CMAKE_SYSTEM_PROCESSOR STREQUAL "x86"   OR
           CMAKE_SYSTEM_PROCESSOR STREQUAL "i386"  OR
           CMAKE_SYSTEM_PROCESSOR STREQUAL "i686"  OR
           CMAKE_SYSTEM_PROCESSOR STREQUAL "AMD32" OR
           CMAKE_SYSTEM_PROCESSOR STREQUAL "amd32" OR
           CMAKE_SYSTEM_PROCESSOR STREQUAL "x32")
        set(OPERATING_SYSTEM_ARCHITECTURE "x32")
    else()
        set(OPERATING_SYSTEM_ARCHITECTURE "x32")
        message(WARNING "Unknown processor architecture for this project project: ${CMAKE_SYSTEM_PROCESSOR}")
        message(WARNING "Set default processor architecture: ${OPERATING_SYSTEM_ARCHITECTURE}")
    endif ()
endif()
