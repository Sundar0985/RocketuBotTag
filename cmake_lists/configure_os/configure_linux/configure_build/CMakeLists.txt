macro(ConfigureDefaultBuildForLinux)
    set(LIBRARY_CURL libcurl.a)
endmacro()

if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    set(CMAKE_C_FLAGS   "${CMAKE_C_FLAGS}   -g -Wall -Wextra -Wconversion -Wsign-conversion -Wpedantic -Weffc++ -Wshadow -pthread")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g -Wall -Wextra -Wconversion -Wsign-conversion -Wpedantic -Weffc++ -Wshadow -pthread")
    set(LIBRARY_CURL libcurl-d.a)
elseif(CMAKE_BUILD_TYPE STREQUAL "Release")
    set(CMAKE_C_FLAGS   "${CMAKE_C_FLAGS}   -O3")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3")
    ConfigureDefaultBuildForLinux()
else()
    ConfigureDefaultBuildForLinux()
    message(WARNING "Unknown build type for this project project: ${CMAKE_BUILD_TYPE}")
    message(WARNING "Set default library for curl: ${LIBRARY_CURL}")
endif()
