macro(ConfigureDefaultBuildForWindows)
    set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded")
    set(LIBRARY_CURL libcurl.lib)
    set(LIBRARY_ZLIB zlib.lib)
endmacro()

if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreadedDebug")
    set(LIBRARY_CURL libcurl-d.lib)
    set(LIBRARY_ZLIB zlibd.lib)
elseif(CMAKE_BUILD_TYPE STREQUAL "Release")
    ConfigureDefaultBuildForWindows()
else()
    ConfigureDefaultBuildForWindows()
    message(WARNING "Unknown build type for this project project: ${CMAKE_BUILD_TYPE}")
    message(WARNING "Set default library for curl: ${LIBRARY_CURL} and zlib: ${LIBRARY_ZLIB}")
    message(WARNING "Set default CMAKE_MSVC_RUNTIME_LIBRARY: ${CMAKE_MSVC_RUNTIME_LIBRARY}")
endif()
